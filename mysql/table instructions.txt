CREATE TABLE characters (
	name VARCHAR(32) PRIMARY KEY,
	type VARCHAR(4),
	numaspects SMALLINT DEFAULT 5,
	skillpoints SMALLINT DEFAULT 20,
	freestunts SMALLINT DEFAULT 3,
	refresh SMALLINT DEFAULT 3,
	slots1 SMALLINT DEFAULT 4,
	slots2 SMALLINT DEFAULT 3,
	slots3 SMALLINT DEFAULT 2,
	slots4 SMALLINT DEFAULT 1,
	slots5 SMALLINT DEFAULT 0,
	slots6 SMALLINT DEFAULT 0,
	slots7 SMALLINT DEFAULT 0,
	slots8 SMALLINT DEFAULT 0,
	cap SMALLINT DEFAULT 4,
	init_physical_bonus SMALLINT DEFAULT 0,
	init_social_bonus SMALLINT DEFAULT 0
) ENGINE InnoDB;
	
CREATE TABLE skills (
	name VARCHAR(32) PRIMARY KEY,
	athletics SMALLINT DEFAULT 0,
	burglary SMALLINT DEFAULT 0,
	contacts SMALLINT DEFAULT 0,
	crafts SMALLINT DEFAULT 0,
	deceive SMALLINT DEFAULT 0,
	drive SMALLINT DEFAULT 0,
	empathy SMALLINT DEFAULT 0,
	fight SMALLINT DEFAULT 0,
	investigate SMALLINT DEFAULT 0,
	lore SMALLINT DEFAULT 0,
	notice SMALLINT DEFAULT 0,
	physique SMALLINT DEFAULT 0,
	provoke SMALLINT DEFAULT 0,
	rapport SMALLINT DEFAULT 0,
	resources SMALLINT DEFAULT 0,
	shoot SMALLINT DEFAULT 0,
	stealth SMALLINT DEFAULT 0,
	will SMALLINT DEFAULT 0
) ENGINE InnoDB;

CREATE TABLE aspects (
	name VARCHAR(32),
	aspect VARCHAR(128),
	num SMALLINT,
	INDEX(name(32)),
	INDEX(aspect(128))
) ENGINE InnoDB;

CREATE TABLE stunts (
	name VARCHAR(32),
	stunt VARCHAR(48),
	description VARCHAR(1024),
	num SMALLINT,
	INDEX(name(32)),
	INDEX(stunt(48))
) ENGINE InnoDB;
